pipeline {
    agent any
    tools {
        maven 'MAVEN_HOME'
    }
    stages {
        stage('Clone & Clean') {
            steps {
                // bat 'rmdir /s /q Maven'
                bat 'git clone https://github.com/Rishi-Kanishka/Maven.git'
                bat 'mvn clean -f Maven'
            }
        }
        stage('install') {
            steps {
                bat 'mvn install -f Maven'
            }
        }
        stage('Test') {
            steps {
                bat 'mvn test -f Maven'
            }
        }
        stage('Package') {
            steps {
                bat 'mvn package -f Maven'
            }
        }
    }
}
